# Spring Cloud 脱离Nacos本地部署方式

1. 在 jar 包同级建立 `config` 文件
2. 新建 bootstrap.yml

```yaml
spring:
  application:
    name: dcip-service
  cloud:
    nacos:
      config:
        enabled: false
    server:
      native:
        search-locations: classpath:/config
```

3. 新建 application.yml, 配置参考如下

   ```yaml
   spring:
     cloud:
       nacos:
         discovery:
           enabled: false
     datasource:
       driver-class-name: com.mysql.cj.jdbc.Driver
       url: jdbc:mysql://192.168.0.201:3306/dcip?characterEncoding=utf8&connectTimeout=3000&socketTimeout=600000&autoReconnect=true&useUnicode=true&allowMultiQueries=true&useSSL=false&serverTimezone=UTC
       username: root
       password: 12345678
       hikari:
         maximum-pool-size: 10
         data-source-properties:
           cachePrepStmts: true
           prepStmtCacheSize: 500
           prepStmtCacheSqlLimit: 2048
           useServerPrepStmts: true
           useLocalSessionState: false
           rewriteBatchedStatements: true
           cacheResultSetMetadata: true
           cacheServerConfiguration: true
           elideSetAutoCommits: true
           maintainTimeStats: false
     mvc:
       servlet:
         path: /api
     servlet:
       multipart:
         max-file-size: 100MB
         max-request-size: 100MB
   
     redis:
       host: ${REDIS_HOST:192.168.0.203}
       port: ${REDIS_PORT:6379}
     rabbitmq:
       host: 192.168.0.203
       port: 5673
       username: dev
       password: dev
       virtual_host: dev
   server:
     servlet:
       context-path: /dcip
     port: 10200
   logging:
     level:
       root: info
       cn.longnows: debug
       org.springframework: WARN
       org.mybatis.spring: WARN
       io.swagger: WARN
       com.baomidou.mybatisplus: WARN
       springfox.documentation: WARN
       com.alibaba.nacos: WARN
       com.alibaba.cloud: WARN
     file:
       path: logs
       name: logs/log.log
     pattern:
       console: "%date level=%level [%thread] [%mdc] %logger - %msg %ex%n"
       file: "%date level=%level [%thread] [%mdc] %logger-[%file : %line] - %msg %ex%n"
     logback:
       rollingpolicy:
         max-file-size: 100MB
   
   feign:
     sentinel:
       enabled: false
   
   longnows:
     pubparam:
       url: http://192.168.0.148/pubparam/api
       appId: appId
       secret: secret
   
   management:
     endpoints:
       web:
         exposure:
           include: "*"
   
   springdoc:
     swagger-ui:
       path: /swagger-ui.html
     api-docs:
       path: /api-docs
   ```
